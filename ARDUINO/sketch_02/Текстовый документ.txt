#include <DHT.h>
#include <ESP8266WiFi.h>

// Настройки Wi‑Fi
const char* ssid = "your_wifi_ssid";        // Замените на имя вашей Wi‑Fi сети
const char* password = "your_wifi_password"; // Замените на пароль вашей Wi‑Fi сети
const char* serverIP = "192.168.1.100";  // Замените на IP вашего компьютера с сервером
const int serverPort = 5000;

// Пины подключения датчиков
#define MQ7_PIN A0
#define MQ135_PIN A1
#define KY028_ANALOG A2
#define KY028_DIGITAL 3
#define DHT_PIN 2

// Тип датчика DHT
#define DHT_TYPE DHT11

// Интервал отправки данных (10 секунд для теста)
const unsigned long SEND_INTERVAL = 10000;
unsigned long lastSendTime = 0;

// Объекты
DHT dht(DHT_PIN, DHT_TYPE);
WiFiClient client;

void setup() {
  Serial.begin(9600);
  dht.begin();
  pinMode(KY028_DIGITAL, INPUT);
  lastSendTime = millis();

  // Подключение к Wi‑Fi
  WiFi.begin(ssid, password);
  Serial.print("Подключение к Wi‑Fi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.print(".");
  }
  Serial.println("\nПодключено к Wi‑Fi!");
  Serial.print("IP адрес: ");
  Serial.println(WiFi.localIP());
  Serial.println("Система мониторинга запущена. Отправка данных каждые 10 секунд.");
}

void loop() {
  unsigned long currentTime = millis();

  if (currentTime - lastSendTime >= SEND_INTERVAL) {
    lastSendTime = currentTime;
    sendDataToServer();
  }
  delay(1000); // Задержка 1 с между проверками
}

void sendDataToServer() {
  // Чтение данных с датчиков
  int mq7Value = analogRead(MQ7_PIN);
  int mq135Value = analogRead(MQ135_PIN);
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature();
  int ky028Analog = analogRead(KY028_ANALOG);
  int ky028Digital = digitalRead(KY028_DIGITAL);

  // Проверка ошибок чтения DHT11
  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("Ошибка: не удалось прочитать данные с DHT11");
    return;
  }

  // Формирование JSON
  String jsonData = "{";
  jsonData += ""mq7":" + String(mq7Value) + ",";
  jsonData += ""mq135":" + String(mq135Value) + ",";
  jsonData += ""temp":" + String(temperature, 2) + ","; // 2 знака после запятой
  jsonData += ""hum":" + String(humidity, 2) + ",";    // 2 знака после запятой
  jsonData += ""ky028_analog":" + String(ky028Analog) + ",";
  jsonData += ""ky028_digital":" + String(ky028Digital);
  jsonData += "}";

  Serial.println("Отправка данных на сервер:");
  Serial.println(jsonData);

  // Попытка подключения к серверу
  if (client.connect(serverIP, serverPort)) {
    // Отправляем HTTP POST‑запрос
    client.println("POST /data HTTP/1.1");
    client.println("Host: " + String(serverIP));
    client.println("Content-Type: application/json");
    client.print("Content-Length: ");
    client.println(jsonData.length());
    client.println();
    client.println(jsonData);

    // Ждём ответа
    unsigned long timeout = millis();
    while (client.available() == 0) {
      if (millis() - timeout > 5000) {
        Serial.println("Таймаут: нет ответа от сервера");
        client.stop();
        return;
      }
    }

    // Читаем ответ сервера
    while (client.available()) {
      String line = client.readStringUntil('\r');
      Serial.print(line);
    }
    client.stop();
  } else {
    Serial.println("Ошибка подключения к серверу");
  }
}
